        async function loadProfessionalRatings(professionalId) {
            try {
                const ratingsSnapshot = await getDocs(
                    query(
                        collection(db, "ratings"),
                        where("professionalId", "==", professionalId)
                    )
                );

                const ratings = [];
                let totalRating = 0;

                ratingsSnapshot.forEach((doc) => {
                    const rating = doc.data();
                    ratings.push(rating);
                    totalRating += rating.rating;
                });

                return {
                    ratings: ratings,
                    averageRating: ratings.length > 0 ? (totalRating / ratings.length) : 0,
                    totalCount: ratings.length
                };
            } catch (error) {
                console.error('ط®ط·ط£ ظپظٹ طھط­ظ…ظٹظ„ ط§ظ„طھظ‚ظٹظٹظ…ط§طھ:', error);

        function maskPhone(phone) {
            if (!phone || phone.length < 10) return phone;
            return phone.substring(0, 4) + 'x'.repeat(phone.length - 4);
        }

        function generateStarsDisplay(rating) {
            let stars = '';
            const fullStars = Math.floor(rating);
            const hasHalfStar = rating - fullStars >= 0.5;

            for (let i = 0; i < 5; i++) {
                if (i < fullStars) {
                    stars += '<span class="star filled">â­گ</span>';
                } else if (i === fullStars && hasHalfStar) {
                    stars += '<span class="star filled">â­گ</span>';
                } else {
                    stars += '<span class="star">â­گ</span>';
                }
            }

            return stars;
        }

        function checkForDataChanges(oldData, newData) {
            if (oldData.length !== newData.length) {
                return true;
            }

            for (let i = 0; i < oldData.length; i++) {
                const oldPro = oldData[i];
                const newPro = newData.find(p => p.id === oldPro.id);

                if (!newPro) return true;

                if (oldPro.name !== newPro.name ||
                    oldPro.job !== newPro.job ||
                    oldPro.status !== newPro.status ||
                    oldPro.phone !== newPro.phone ||
                    oldPro.city !== newPro.city ||
                    oldPro.hourlyPrice !== newPro.hourlyPrice ||
                    oldPro.ratings?.totalCount !== newPro.ratings?.totalCount ||
                    Math.abs((oldPro.ratings?.averageRating || 0) - (newPro.ratings?.averageRating || 0)) > 0.1) {
                    return true;
                }
            }

            return false;
        }

window.toggleHamburgerMenu = function() {
    const menu = document.getElementById('hamburgerMenu');
    const overlay = document.getElementById('hamburgerOverlay');
    
    menu.classList.toggle('show');
    overlay.classList.toggle('show');
    
    // Update favorites count in hamburger

}

window.closeHamburgerMenu = function() {
    const menu = document.getElementById('hamburgerMenu');
    const overlay = document.getElementById('hamburgerOverlay');
    
    menu.classList.remove('show');
    overlay.classList.remove('show');
}

}

function updateHamburgerFavorites() {
    const hamburgerFav = document.getElementById('hamburgerFavorites');
    const hamburgerAuth = document.getElementById('hamburgerAuth');
    const hamburgerFavCount = document.getElementById('hamburgerFavCount');
    
    if (currentUser) {
        // Show favorites, hide auth
        if (hamburgerFav) hamburgerFav.style.display = 'flex';
        if (hamburgerAuth) {
            hamburgerAuth.querySelector('.link-text').textContent = 'طھط³ط¬ظٹظ„ ط®ط±ظˆط¬';
            hamburgerAuth.querySelector('.link-icon').textContent = 'ًںڑھ';
            hamburgerAuth.onclick = (e) => {
                e.preventDefault();
                logout();
                closeHamburgerMenu();
            };
            hamburgerAuth.removeAttribute('href');
        }
        
        // Update count
        if (hamburgerFavCount) {
            hamburgerFavCount.textContent = favoritesCount;
        }
    } else {
        // Hide favorites, show auth
        if (hamburgerFav) hamburgerFav.style.display = 'none';
        if (hamburgerAuth) {
            hamburgerAuth.querySelector('.link-text').textContent = 'طھط³ط¬ظٹظ„ ط§ظ„ط¯ط®ظˆظ„';
            hamburgerAuth.querySelector('.link-icon').textContent = 'ًں”گ';
            hamburgerAuth.onclick = null;
            hamburgerAuth.href = 'customer-auth-page.html';
        }
    }
}
